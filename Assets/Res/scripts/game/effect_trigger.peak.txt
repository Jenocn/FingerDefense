import "Game";

enum BrickType {
	Bomb = 4,
}

enum EffectType {
	BoomArea = 1, // 圆形范围炸弹
	BrickDie = 2, // 砖块死亡特效
}

enum ElementType {
	None = 0,
	Ball, // 球
	Brick, // 砖块
	Racket, // 球拍
	Edge, // 边缘
	Bomb, // 炸弹
}

// 砖块被击时
function brick_hit(brickID, x, y, attackID, attackElementType) {
}

// 砖块死亡时
function brick_die(brickID, x, y, attackID, attackElementType) {
	if (brickID == BrickType.Bomb) {
		create_effect(EffectType.BoomArea, x, y, 0.1);
	}
	create_effect(EffectType.BrickDie, x, y, 0);
}
